#include <stdio.h>

int min(int a, int b){
    return a > b ? b : a;
}
int max(int a, int b){
    return a > b ? a : b;
}

int main(){
    int n, m, maxcur, cnt = 0;
    scanf("%d%d", &m, &n);
    int matrix[256][256];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            scanf("%d", &matrix[i][j]);

    for(int i=0; i < n; i++){
        for(int j=0; j < m; j++){
            maxcur = matrix[i][j];

            for (int p=0; p < n - i; p++)

                for(int p1 = max(j-p, 0); p1 < min(j+p, m); p1++)
                    maxcur = maxcur > matrix[p+i][p1] ? maxcur : matrix[p+i][p1];

            cnt += (matrix[i][j] == maxcur);
            matrix[i][j] = maxcur;
        }
    }

    for(int i=0; i < n; i++) {
        for (int j = 0; j < m; j++)
            printf("%d ", matrix[i][j]);
        printf("\n");
    }
}
